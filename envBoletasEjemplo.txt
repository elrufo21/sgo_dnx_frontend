     public void calculaFechaReferencia()
        {
            xFechaRefencia = string.Empty;
            xRangoNumeros = string.Empty;
            string xdia = string.Empty;
            string[] xD1;
            string[] xD2;
            string xinicial = string.Empty;
            string xserie, xserieB = string.Empty;
            int xn1 = 0;
            int xn2 = 0;
            int indiceUltimaFila = gvlista.Rows.Count - 1;
            DateTime dia2;
            for (int i = 0; i <= 0; i++)
            {
                xdia = Convert.ToString(gvlista.Rows[i].Cells[3].Value);
                xinicial = Convert.ToString(gvlista.Rows[i].Cells[5].Value);
            }
            xD1 = xinicial.Split('-');
            xserie = xD1[0].Replace("0", "");
            xn1 = int.Parse(xD1[1]);
            xRangoNumeros = xserie + "-" + xn1.ToString();
            xD2 = gvlista.Rows[indiceUltimaFila].Cells[5].Value.ToString().Split('-');
            xserieB = xD2[0].Replace("0", "");
            xn2 = int.Parse(xD2[1]);
            xRangoNumeros += " al " + xserieB + "-" + xn2.ToString();
            dia2 = Convert.ToDateTime(xdia);
            xFechaRefencia = dia2.ToString("yyyy-MM-dd");
        }
